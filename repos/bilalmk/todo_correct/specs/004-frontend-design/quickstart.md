# Quick Start Guide: Modern Frontend Design System

**Feature**: 004-frontend-design
**Date**: 2025-12-31
**Phase**: Phase 1 - Design & Contracts

## Overview

This guide provides step-by-step instructions to set up and implement the modern frontend design system for the Todo Evolution application. The implementation is UI-only (no backend integration) and uses Next.js 16+ App Router, TypeScript, Tailwind CSS, shadcn/ui, and Framer Motion.

## Prerequisites

Before starting, ensure you have:

- **Node.js**: 18.x or higher (check: `node --version`)
- **npm** or **pnpm**: Latest stable version
- **Git**: For version control
- **Code Editor**: VS Code recommended with TypeScript/Tailwind extensions
- **Existing Next.js Project**: The frontend/ directory should already be initialized from previous phases

## Step 1: Install Dependencies

Navigate to the `frontend/` directory and install all required packages:

```bash
cd frontend

# Core dependencies
npm install framer-motion react-hook-form @hookform/resolvers zod react-day-picker lucide-react react-colorful

# shadcn/ui initialization (if not already done)
npx shadcn@latest init

# Add shadcn/ui components
npx shadcn@latest add button card input form dialog badge tabs checkbox select popover calendar sonner skeleton

# Drag-and-drop library (visual only)
npm install @dnd-kit/core @dnd-kit/sortable

# Date formatting utility (optional)

```

## Step 5: Create Animation Variants

Create `src/lib/animations.ts`:

```typescript
import { Variants } from "framer-motion"

// Check for reduced motion preference
const prefersReducedMotion =
  typeof window !== "undefined" &&
  window.matchMedia("(prefers-reduced-motion: reduce)").matches

const getTransition = (defaultTransition: any) => {
  if (prefersReducedMotion) {
    return { duration: 0 }
  }
  return defaultTransition
}

// Short (150ms) - Micro-interactions
export const fadeIn: Variants = {
  initial: { opacity: 0 },
  animate: { opacity: 1 },
  exit: { opacity: 0 },
  transition: getTransition({ type: "spring", stiffness: 300, damping: 30 }),
}

// Medium (300ms) - Standard transitions
export const slideUp: Variants = {
  initial: { opacity: 0, y: 20 },
  animate: { opacity: 1, y: 0 },
  exit: { opacity: 0, y: -20 },
  transition: getTransition({ type: "spring", stiffness: 200, damping: 25 }),
}

export const modalScale: Variants = {
  initial: { opacity: 0, scale: 0.95 },
  animate: { opacity: 1, scale: 1 },
  exit: { opacity: 0, scale: 0.95 },
  transition: getTransition({ type: "spring", stiffness: 300, damping: 30 }),
}

// List item animation
export const listItem: Variants = {
  initial: { opacity: 0, x: -20 },
  animate: { opacity: 1, x: 0 },
  exit: { opacity: 0, x: 20 },
  transition: getTransition({ type: "spring", stiffness: 250, damping: 25 }),
}

// Long (500ms) - Page-level changes
export const pageTransition: Variants = {
  initial: { opacity: 0, y: 10 },
  animate: { opacity: 1, y: 0 },
  exit: { opacity: 0, y: -10 },
  transition: getTransition({ type: "spring", stiffness: 150, damping: 20 }),
}
```

## Step 8: Create React Context Providers

### TaskContext

Create `src/contexts/TaskContext.tsx`:

```typescript
"use client"

import { createContext, useContext, useState, useEffect, ReactNode } from "react"
import { Task, CreateTaskInput, createTask, updateTask as updateTaskHelper } from "@/types/task"
import { MOCK_TASKS, delay } from "@/lib/mock-data"

interface TaskContextValue {
  tasks: Task[]
  isLoading: boolean
  addTask: (input: CreateTaskInput) => Promise<void>
  updateTask: (id: string, updates: Partial<Task>) => Promise<void>
  deleteTask: (id: string) => Promise<void>
  completeTask: (id: string, completed: boolean) => Promise<void>
}

const TaskContext = createContext<TaskContextValue | undefined>(undefined)

export function TaskProvider({ children }: { children: ReactNode }) {
  const [tasks, setTasks] = useState<Task[]>([])
  const [isLoading, setIsLoading] = useState(true)

  // Load from localStorage on mount
  useEffect(() => {
    const stored = localStorage.getItem("todo_app_tasks")
    if (stored) {
      setTasks(JSON.parse(stored))
    } else {
      setTasks(MOCK_TASKS)
    }
    setIsLoading(false)
  }, [])

  // Sync to localStorage on tasks change
  useEffect(() => {
    if (!isLoading) {
      localStorage.setItem("todo_app_tasks", JSON.stringify(tasks))
    }
  }, [tasks, isLoading])

  const addTask = async (input: CreateTaskInput) => {
    await delay(Math.random() * 500 + 300) // 300-800ms
    const newTask = createTask(input)
    setTasks((prev) => [newTask, ...prev])
  }

  const updateTask = async (id: string, updates: Partial<Task>) => {
    await delay(Math.random() * 500 + 300)
    setTasks((prev) =>
      prev.map((task) =>
        task.id === id ? updateTaskHelper(task, updates) : task
      )
    )
  }

  const deleteTask = async (id: string) => {
    await delay(Math.random() * 500 + 300)
    setTasks((prev) => prev.filter((task) => task.id !== id))
  }

  const completeTask = async (id: string, completed: boolean) => {
    await delay(200)
    setTasks((prev) =>
      prev.map((task) =>
        task.id === id
          ? { ...task, completed, updated_at: new Date().toISOString() }
          : task
      )
    )
  }

  return (
    <TaskContext.Provider
      value={{ tasks, isLoading, addTask, updateTask, deleteTask, completeTask }}
    >
      {children}
    </TaskContext.Provider>
  )
}

export function useTasks() {
  const context = useContext(TaskContext)
  if (!context) {
    throw new Error("useTasks must be used within TaskProvider")
  }
  return context
}
```

## Troubleshooting

### Issue: Next.js 16 params/searchParams errors

**Solution**: Ensure all page components use `await params` (Server Components) or `use(params)` (Client Components).

### Issue: Tailwind styles not applying

**Solution**: Check `tailwind.config.ts` content paths include your files. Restart dev server.

### Issue: shadcn/ui components not found

**Solution**: Run `npx shadcn@latest init` and ensure components are in `src/components/ui/`.

### Issue: Framer Motion hydration errors

**Solution**: Ensure animations are wrapped in client components (`"use client"`).
